def validate_position(self, symbol: str, margin: float, account_capital: float = None) -> bool:
    """验证仓位大小"""
    # 检查日度风险指标
    self.reset_daily_metrics()
    
    # 检查保证金限制
    total_margin_used = sum(pos.get('margin', 0.0) for pos in self.positions.values())

    return True
